/*Program to add, subtract, multiply,and Divide two numbers*/

#include< stdio.h >
void main()
{
int a,b,add,sub,mul;
float div;
printf("Enter two numbers:");
scanf("%d%d",&a,&b);
add=a+b;
sub=a-b;
mul=a*b;
div=a/b;
printf("\nAddition is :%d",add);
printf("\nSubtraction is :%d",sub);
printf("\nMultiplication is :%d",mul);
printf("\nDivision is :%0.3f",div);
}
