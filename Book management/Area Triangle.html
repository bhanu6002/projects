/*Program to calculate area of triangle*/

#include< stdio.h >
void main()
{
int h,b;
float area;
printf("Enter height and base of triangle:");
scanf("%d%d",&h,&b);
area=0.5*h*b;
printf("The Area of Triangle is %0.3f",area);
}
